20inline String resolve_model_path(const String &p_path) {
21 if (p_path.is_empty()) {
22 return "";
23 }
24
25 ProjectSettings *ps = ProjectSettings::get_singleton();
26 PackedStringArray suffixes;
27 if (ps && ps->has_setting("gaze/models/acceptable_suffixes")) {
28 suffixes = ps->get_setting("gaze/models/acceptable_suffixes");
29 } else {
30 suffixes.push_back(".ort");
31 suffixes.push_back(".xml");
32 }
33
34 // Get the base prefix by stripping any trailing extension if present.
35 String prefix = p_path.get_basename();
36 String filename_prefix = prefix.get_file();
37
38 // 1. Direct path checking with suffixes (for res:// or user:// absolute paths)
39 if (p_path.begins_with("res://") || p_path.begins_with("user://")) {
40 for (int j = 0; j < suffixes.size(); j++) {
41 String candidate = prefix + suffixes[j];
42 if (FileAccess::file_exists(candidate)) {
43 return candidate;
44 }
45 }
46 if (FileAccess::file_exists(p_path)) {
47 return p_path;
48 }
49 }
50
51 // 2. Check sequentially in search paths
52 String search_paths_str = "res://models,res://addons/godot-gaze/models";
53 if (ps && ps->has_setting("gaze/models/search_paths")) {
54 search_paths_str = ps->get_setting("gaze/models/search_paths");
55 }
56
57 PackedStringArray paths = search_paths_str.split(",");
58 for (int i = 0; i < paths.size(); i++) {
59 String dir = paths[i].strip_edges();
60 if (dir.is_empty()) continue;
61 if (!dir.ends_with("/")) {
62 dir += "/";
63 }
64 for (int j = 0; j < suffixes.size(); j++) {
65 String candidate = dir + filename_prefix + suffixes[j];
66 if (FileAccess::file_exists(candidate)) {
67 return candidate;
68 }
69 }
70 String exact_candidate = dir + p_path.get_file();
71 if (FileAccess::file_exists(exact_candidate)) {
72 return exact_candidate;
73 }
74 }
75
76 // 3. Direct prefix and path fallbacks
77 for (int j = 0; j < suffixes.size(); j++) {
78 String candidate = prefix + suffixes[j];
79 if (FileAccess::file_exists(candidate)) {
80 return candidate;
81 }
82 }
83 if (FileAccess::file_exists(p_path)) {
84 return p_path;
85 }
86
87 return p_path;
88}

93inline std::vector<uint8_t> load_file_buffer(const String &path) {
94 std::vector<uint8_t> buffer;
95 if (path.is_empty()) {
96 return buffer;
97 }
98 Ref<FileAccess> file = FileAccess::open(path, FileAccess::READ);
99 if (file.is_null()) {
100 Gaze::log_error("LoadFileBufferFailed_Open", "path", path.utf8().get_data());
101 return buffer;
102 }
103 uint64_t length = file->get_length();
104 if (length == 0) {
105 Gaze::log_error("LoadFileBufferFailed_Empty", "path", path.utf8().get_data());
106 return buffer;
107 }
108 PackedByteArray godot_buffer = file->get_buffer(length);
109 buffer.resize(length);
110 std::memcpy(buffer.data(), godot_buffer.ptr(), length);
111 return buffer;
112}
