81 GazeVector2 get_pitch_yaw(const GazeVector3& relative = GazeVector3(0, 0, -1)) const {
82 GazeVector3 n = normalized();
83 double pitch_rad = std::asin(n.y);
84
85 double yaw_rad = std::atan2(-n.x, -n.z);
86 double rel_yaw_rad = std::atan2(-relative.x, -relative.z);
87 double yaw_diff = yaw_rad - rel_yaw_rad;
88 while (yaw_diff > PI) yaw_diff -= TAU;
89 while (yaw_diff < -PI) yaw_diff += TAU;
90
91 return GazeVector2(
92 pitch_rad * RAD_TO_DEG,
93 yaw_diff * RAD_TO_DEG
94 );

118 static GazeBasis3D from_euler_zyx(double pitch_deg, double yaw_deg, double roll_deg) {
119 double p = pitch_deg * DEG_TO_RAD;
120 double y = yaw_deg * DEG_TO_RAD;
121 double r = roll_deg * DEG_TO_RAD;
122 double cp = std::cos(p), sp = std::sin(p);
123 double cy = std::cos(y), sy = std::sin(y);
124 double cr = std::cos(r), sr = std::sin(r);
125 return GazeBasis3D(
126 GazeVector3(cr * cy, sr * cy, -sy),
127 GazeVector3(cr * sy * sp - sr * cp, sr * sy * sp + cr * cp, cy * sp),
128 GazeVector3(cr * sy * cp + sr * sp, sr * sy * cp - cr * sp, cy * cp)
129 );
130 }

142 GazeVector3 get_euler_deg() const {
143 double sy = std::sqrt(x.x * x.x + x.y * x.y);
144 bool singular = sy < 1e-6;
145 double pitch = 0.0, yaw = 0.0, roll = 0.0;
146 if (!singular) {
147 // Since the face points towards the camera, yaw is around 180 degrees.
148 // This means cos(yaw) < 0. We decompose choosing the branch where cos(yaw) is negative.
149 pitch = std::atan2(-y.z, -z.z) * RAD_TO_DEG;
150 yaw = std::atan2(-x.z, -sy) * RAD_TO_DEG;
151 roll = std::atan2(-x.y, -x.x) * RAD_TO_DEG;
152 } else {
153 pitch = std::atan2(-y.y, y.x) * RAD_TO_DEG;
154 yaw = std::atan2(-x.z, -sy) * RAD_TO_DEG;
155 roll = 0.0;
156 }
157 return GazeVector3(pitch, yaw, roll);
158 }

207inline GazeBasis3D rodrigues_to_basis(const GazeVector3& r) {
208 double theta = r.length();
209 if (theta < 1e-6) {
210 return GazeBasis3D();
211 }
212 double ux = r.x / theta;
213 double uy = r.y / theta;
214 double uz = r.z / theta;
215
216 double s = std::sin(theta);
217 double c = std::cos(theta);
218 double oc = 1.0 - c;
219
220 return GazeBasis3D(
221 GazeVector3(c + oc * ux * ux, oc * ux * uy + s * uz, oc * ux * uz - s * uy),
222 GazeVector3(oc * ux * uy - s * uz, c + oc * uy * uy, oc * uy * uz + s * ux),
223 GazeVector3(oc * ux * uz + s * uy, oc * uy * uz - s * ux, c + oc * uz * uz)
224 );
225}

228inline GazeVector3 get_head_forward_in_camera_space(const GazeVector3& rotation_deg) {
229 double pitch_rad = rotation_deg.x * DEG_TO_RAD;
230 double yaw_rad = rotation_deg.y * DEG_TO_RAD;
231 double roll_rad = rotation_deg.z * DEG_TO_RAD;
232
233 double cp = std::cos(pitch_rad), sp = std::sin(pitch_rad);
234 double cy = std::cos(yaw_rad), sy = std::sin(yaw_rad);
235 double cr = std::cos(roll_rad), sr = std::sin(roll_rad);
236
237 double r02 = cr * sy * cp + sr * sp;
238 double r12 = sr * sy * cp - cr * sp;
239 double r22 = cy * cp;
240
241 return GazeVector3(-r02, r12, r22);
242}
